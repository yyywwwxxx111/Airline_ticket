USE Project;
CREATE TABLE IF NOT EXISTS flight_ticket(
date_day VARCHAR(20),
company VARCHAR(30),
plane VARCHAR(20),
depart_city VARCHAR(20),
destination VARCHAR(20),
flight_number VARCHAR(20),
depart_time VARCHAR(20),
depart_airport VARCHAR(20),
depart_terminal VARCHAR(10),
duration VARCHAR(20),
arrival_time VARCHAR(20),
arrival_airport VARCHAR(20),
arrival_terminal VARCHAR(10),
price VARCHAR(20),
annotation VARCHAR(40),
isdeleted INT,
id SERIAL PRIMARY KEY NOT NULL
)
alter table flight_ticket convert to character set utf8mb4;
CREATE TABLE IF NOT EXISTS company(
id SERIAL PRIMARY KEY NOT NULL,
NAME VARCHAR(30)
);
alter table company convert to character set utf8mb4;
